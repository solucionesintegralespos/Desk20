INSERT INTO sync_data_tbl (
    sync_data_id,
    data_key,
    object_name,
    date,
    is_sent,
    is_deleted
)
SELECT  
NEWID(),
order_id,
'DAL.order_tbl',
GETDATE(),
0,
0
FROM Order_tbl o
WHERE NOT EXISTS (
    SELECT 1
    FROM sync_data_tbl s
    WHERE s.data_key = o.order_id
      AND s.object_name LIKE '%Order_tbl'
)
